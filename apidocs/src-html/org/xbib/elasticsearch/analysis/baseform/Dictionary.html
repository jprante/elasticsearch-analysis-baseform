002package org.xbib.elasticsearch.analysis.baseform;
003
004import java.io.BufferedReader;
005import java.io.IOException;
006import java.io.InputStreamReader;
007import java.io.Reader;
008import java.nio.ByteBuffer;
009import java.nio.CharBuffer;
010import java.nio.charset.CharacterCodingException;
011import java.nio.charset.Charset;
012import java.util.ArrayList;
013import java.util.Collections;
014import java.util.List;
015
016import static org.xbib.elasticsearch.analysis.baseform.MatchResult.EXACT_MATCH;
017import static org.xbib.elasticsearch.analysis.baseform.MatchResult.NO_MATCH;
018import static org.xbib.elasticsearch.analysis.baseform.MatchResult.SEQUENCE_IS_A_PREFIX;
019
020public class Dictionary {
021
022    private final Charset UTF8 = Charset.forName("UTF-8");
023
024    private FSA fsa;
025
026    private FSATraversal matcher;
027
028    public Dictionary load(String language) throws IOException {
029        return load(new InputStreamReader(this.getClass().getResourceAsStream(language+"-lemma-utf8.txt"),UTF8));
030    }
031
032    public Dictionary load(Reader in) throws IOException {
033        BufferedReader reader = new BufferedReader(in);
034        List<byte[]> lines = new ArrayList<byte[]>();
035        String line;
036        while ((line = reader.readLine()) != null) {
037            lines.add(line.replace('\t', '+').getBytes(UTF8));
038        }
039        reader.close();
040        Collections.sort(lines, FSABuilder.LEXICAL_ORDERING);
041        FSABuilder builder = new FSABuilder();
042        for (byte[] b : lines) {
043            builder.add(b, 0, b.length);
044        }
045        this.fsa = builder.complete();
046        this.matcher = new FSATraversal(fsa);
047        return this;
048    }
049
050    public String lookup(CharSequence prefix) throws CharacterCodingException {
051        return lookup(UTF8.newEncoder().encode(CharBuffer.wrap(prefix)), prefix.toString());
052    }
053
054    public String lookup(ByteBuffer buf, String result) {
055        MatchResult match = matcher.match(buf.array(), buf.position(), buf.remaining(), fsa.getRootNode());
056        switch (match.kind) {
057            case SEQUENCE_IS_A_PREFIX: {
058                final int arc = fsa.getArc(match.node, (byte)'+');
059                if (arc != 0 && !fsa.isArcFinal(arc)) {
060                    FSAFinalStatesIterator finalStatesIterator = new FSAFinalStatesIterator(fsa, fsa.getRootNode());
061                    finalStatesIterator.restartFrom(fsa.getEndNode(arc));
062                    if (finalStatesIterator.hasNext()) {
063                        buf = finalStatesIterator.next();
064                        String s = new String(buf.array(), buf.position(), buf.remaining(), UTF8);
065                        return s.isEmpty() || s.equals(result) ? s : lookup(buf, s);
066                    }
067                }
068                break;
069            }
070            case EXACT_MATCH: {
071                break;
072            }
073            case NO_MATCH: {
074                break;
075            }
076        }
077        return result;
078    }
079
080}
