002package org.xbib.elasticsearch.index.analysis;
003
004import org.apache.lucene.analysis.Token;
005import org.apache.lucene.analysis.TokenFilter;
006import org.apache.lucene.analysis.TokenStream;
007import org.apache.lucene.analysis.tokenattributes.CharTermAttribute;
008import org.apache.lucene.analysis.tokenattributes.OffsetAttribute;
009import org.apache.lucene.analysis.tokenattributes.PositionIncrementAttribute;
010import org.apache.lucene.util.AttributeSource;
011
012import org.xbib.elasticsearch.analysis.baseform.Dictionary;
013
014import java.io.IOException;
015import java.nio.charset.CharacterCodingException;
016import java.util.LinkedList;
017
018public class BaseformTokenFilter extends TokenFilter {
019
020    private final LinkedList<Token> tokens;
021
022    private final Dictionary dictionary;
023
024    private final CharTermAttribute termAtt = addAttribute(CharTermAttribute.class);
025
026    private final OffsetAttribute offsetAtt = addAttribute(OffsetAttribute.class);
027
028    private final PositionIncrementAttribute posIncAtt = addAttribute(PositionIncrementAttribute.class);
029
030    private AttributeSource.State current;
031
032    protected BaseformTokenFilter(TokenStream input, Dictionary dictionary) {
033        super(input);
034        this.tokens = new LinkedList<Token>();
035        this.dictionary = dictionary;
036    }
037
038    @Override
039    public final boolean incrementToken() throws IOException {
040        if (!tokens.isEmpty()) {
041            assert current != null;
042            Token token = tokens.removeFirst();
043            restoreState(current);
044            termAtt.setEmpty().append(token);
045            offsetAtt.setOffset(token.startOffset(), token.endOffset());
046            posIncAtt.setPositionIncrement(0);
047            return true;
048        }
049        if (input.incrementToken()) {
050            baseform();
051            if (!tokens.isEmpty()) {
052                current = captureState();
053            }
054            return true;
055        } else {
056            return false;
057        }
058    }
059
060    protected void baseform() throws CharacterCodingException {
061        int start = offsetAtt.startOffset();
062        CharSequence term = new String(termAtt.buffer(), 0, termAtt.length());
063        String s = dictionary.lookup(term);
064        tokens.add(new Token(s, start, start + s.length()));
065    }
066
067    @Override
068    public void reset() throws IOException {
069        super.reset();
070        tokens.clear();
071        current = null;
072    }
073
074}
