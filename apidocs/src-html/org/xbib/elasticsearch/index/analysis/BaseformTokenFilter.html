002    package org.xbib.elasticsearch.index.analysis;
003    
004    import org.apache.lucene.analysis.Token;
005    import org.apache.lucene.analysis.TokenFilter;
006    import org.apache.lucene.analysis.TokenStream;
007    import org.apache.lucene.analysis.tokenattributes.CharTermAttribute;
008    import org.apache.lucene.analysis.tokenattributes.OffsetAttribute;
009    import org.apache.lucene.analysis.tokenattributes.PositionIncrementAttribute;
010    import org.apache.lucene.util.AttributeSource;
011    
012    import org.xbib.elasticsearch.analysis.baseform.Dictionary;
013    
014    import java.io.IOException;
015    import java.nio.charset.CharacterCodingException;
016    import java.util.LinkedList;
017    
018    public class BaseformTokenFilter extends TokenFilter {
019    
020        private final LinkedList<Token> tokens;
021    
022        private final Dictionary dictionary;
023    
024        private final CharTermAttribute termAtt = addAttribute(CharTermAttribute.class);
025    
026        private final OffsetAttribute offsetAtt = addAttribute(OffsetAttribute.class);
027    
028        private final PositionIncrementAttribute posIncAtt = addAttribute(PositionIncrementAttribute.class);
029    
030        private AttributeSource.State current;
031    
032        protected BaseformTokenFilter(TokenStream input, Dictionary dictionary) {
033            super(input);
034            this.tokens = new LinkedList<Token>();
035            this.dictionary = dictionary;
036        }
037    
038        @Override
039        public final boolean incrementToken() throws IOException {
040            if (!tokens.isEmpty()) {
041                assert current != null;
042                Token token = tokens.removeFirst();
043                restoreState(current);
044                termAtt.setEmpty().append(token);
045                offsetAtt.setOffset(token.startOffset(), token.endOffset());
046                posIncAtt.setPositionIncrement(0);
047                return true;
048            }
049            if (input.incrementToken()) {
050                injectBaseform();
051                if (!tokens.isEmpty()) {
052                    current = captureState();
053                }
054                return true;
055            } else {
056                return false;
057            }
058        }
059    
060        protected void injectBaseform() throws CharacterCodingException {
061            int start = offsetAtt.startOffset();
062            CharSequence term = new String(termAtt.buffer(), 0, termAtt.length());
063            CharSequence s = dictionary.lookup(term);
064            if (s != null) {
065                tokens.add(new Token(s.toString(), start, start + s.length()));
066            }
067        }
068    
069        @Override
070        public void reset() throws IOException {
071            super.reset();
072            tokens.clear();
073            current = null;
074        }
075    
076        public void close() throws IOException {
077            super.close();
078            tokens.clear();
079            current = null;
080        }
081    
082    }
