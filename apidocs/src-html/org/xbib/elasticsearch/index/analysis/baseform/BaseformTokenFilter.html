001package org.xbib.elasticsearch.index.analysis.baseform;
002
003import org.apache.lucene.analysis.TokenFilter;
004import org.apache.lucene.analysis.TokenStream;
005import org.apache.lucene.analysis.tokenattributes.CharTermAttribute;
006import org.apache.lucene.analysis.tokenattributes.OffsetAttribute;
007import org.apache.lucene.analysis.tokenattributes.PackedTokenAttributeImpl;
008import org.apache.lucene.analysis.tokenattributes.PositionIncrementAttribute;
009import org.apache.lucene.util.AttributeSource;
010
011import java.io.IOException;
012import java.nio.charset.CharacterCodingException;
013import java.util.LinkedList;
014
015public class BaseformTokenFilter extends TokenFilter {
016
017    private final LinkedList<PackedTokenAttributeImpl> tokens;
018
019    private final Dictionary dictionary;
020
021    private final CharTermAttribute termAtt = addAttribute(CharTermAttribute.class);
022
023    private final OffsetAttribute offsetAtt = addAttribute(OffsetAttribute.class);
024
025    private final PositionIncrementAttribute posIncAtt = addAttribute(PositionIncrementAttribute.class);
026
027    private AttributeSource.State current;
028
029    protected BaseformTokenFilter(TokenStream input, Dictionary dictionary) {
030        super(input);
031        this.tokens = new LinkedList<PackedTokenAttributeImpl>();
032        this.dictionary = dictionary;
033    }
034
035    @Override
036    public final boolean incrementToken() throws IOException {
037        if (!tokens.isEmpty()) {
038            assert current != null;
039            PackedTokenAttributeImpl token = tokens.removeFirst();
040            restoreState(current);
041            termAtt.setEmpty().append(token);
042            offsetAtt.setOffset(token.startOffset(), token.endOffset());
043            posIncAtt.setPositionIncrement(0);
044            return true;
045        }
046        if (input.incrementToken()) {
047            baseform();
048            if (!tokens.isEmpty()) {
049                current = captureState();
050            }
051            return true;
052        } else {
053            return false;
054        }
055    }
056
057    protected void baseform() throws CharacterCodingException {
058        CharSequence term = new String(termAtt.buffer(), 0, termAtt.length());
059        CharSequence s = dictionary.lookup(term);
060        if (s != null && s.length() > 0) {
061            PackedTokenAttributeImpl impl = new PackedTokenAttributeImpl();
062            impl.append(s);
063            tokens.add(impl);
064        }
065    }
066
067    @Override
068    public void reset() throws IOException {
069        super.reset();
070        tokens.clear();
071        current = null;
072    }
073
074    public void close() throws IOException {
075        super.close();
076        tokens.clear();
077        current = null;
078    }
079
080}
