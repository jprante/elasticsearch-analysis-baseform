001package org.xbib.elasticsearch.index.analysis.baseform;
002
003/**
004 * This class implements some common matching and scanning operations on a
005 * generic FSA.
006 */
007public final class FSATraversal {
008    /**
009     * Target automaton.
010     */
011    private final FSA fsa;
012
013    /**
014     * Traversals of the given FSA.
015     */
016    public FSATraversal(FSA fsa) {
017        this.fsa = fsa;
018    }
019
020    /**
021     * Calculate perfect hash for a given input sequence of bytes. The perfect hash requires
022     * that {@link FSA} is built with {@link FSAFlags#NUMBERS} and corresponds to the sequential
023     * order of input sequences used at automaton construction time.
024     *
025     * @param start  Start index in the sequence array.
026     * @param length Length of the byte sequence, must be at least 1.
027     * @return Returns a unique integer assigned to the input sequence in the automaton (reflecting
028     * the number of that sequence in the input used to build the automaton). Returns a negative
029     * integer if the input sequence was not part of the input from which the automaton was created.
030     * The type of mismatch is a constant defined in {@link MatchResult}.
031     */
032    public int perfectHash(byte[] sequence, int start, int length, int node) {
033        assert fsa.getFlags().contains(FSAFlags.NUMBERS) : "FSA not built with NUMBERS option.";
034        assert length > 0 : "must be a non-empty sequence";
035        int hash = 0;
036        final int end = start + length - 1;
037        int seqIndex = start;
038        byte label = sequence[seqIndex];
039
040        // Seek through the current node's labels, looking for 'label', update hash.
041        for (int arc = fsa.getFirstArc(node); arc != 0; ) {
042            if (fsa.getArcLabel(arc) == label) {
043                if (fsa.isArcFinal(arc)) {
044                    if (seqIndex == end) {
045                        return hash;
046                    }
047
048                    hash++;
049                }
050
051                if (fsa.isArcTerminal(arc)) {
052                    /* The automaton contains a prefix of the input sequence. */
053                    return MatchResult.AUTOMATON_HAS_PREFIX;
054                }
055
056                // The sequence is a prefix of one of the sequences stored in the automaton.
057                if (seqIndex == end) {
058                    return MatchResult.SEQUENCE_IS_A_PREFIX;
059                }
060
061                // Make a transition along the arc, go the target node's first arc.
062                arc = fsa.getFirstArc(fsa.getEndNode(arc));
063                label = sequence[++seqIndex];
064                continue;
065            } else {
066                if (fsa.isArcFinal(arc)) {
067                    hash++;
068                }
069                if (!fsa.isArcTerminal(arc)) {
070                    hash += fsa.getRightLanguageCount(fsa.getEndNode(arc));
071                }
072            }
073
074            arc = fsa.getNextArc(arc);
075        }
076
077        // Labels of this node ended without a match on the sequence.
078        // Perfect hash does not exist.
079        return MatchResult.NO_MATCH;
080    }
081
082    /**
083     * @see #perfectHash(byte[], int, int, int)
084     */
085    public int perfectHash(byte[] sequence) {
086        return perfectHash(sequence, 0, sequence.length, fsa.getRootNode());
087    }
088
089    /**
090     * Same as {@link #match(byte[], int, int, int)}, but allows passing
091     * a reusable {@link MatchResult} object so that no intermediate garbage is
092     * produced.
093     *
094     * @return The same object as <code>result</code>, but with reset internal
095     * type and other fields.
096     */
097    public MatchResult match(MatchResult result,
098                             byte[] sequence, int start, int length, int node) {
099        if (node == 0) {
100            result.reset(MatchResult.NO_MATCH, start, node);
101            return result;
102        }
103        final FSA fsa = this.fsa;
104        final int end = start + length;
105        for (int i = start; i < end; i++) {
106            final int arc = fsa.getArc(node, sequence[i]);
107            if (arc != 0) {
108                if (fsa.isArcFinal(arc) && i + 1 == end) {
109                    /* The automaton has an exact match of the input sequence. */
110                    result.reset(MatchResult.EXACT_MATCH, i, node);
111                    return result;
112                }
113                if (fsa.isArcTerminal(arc)) {
114                                        /* The automaton contains a prefix of the input sequence. */
115                    result.reset(MatchResult.AUTOMATON_HAS_PREFIX, i + 1, 0);
116                    return result;
117                }
118                // Make a transition along the arc.
119                node = fsa.getEndNode(arc);
120            } else {
121                result.reset(MatchResult.NO_MATCH, i, node);
122                return result;
123            }
124        }
125                /* The sequence is a prefix of at least one sequence in the automaton. */
126        result.reset(MatchResult.SEQUENCE_IS_A_PREFIX, 0, node);
127        return result;
128    }
129
130    /**
131     * Finds a matching path in the dictionary for a given sequence of labels
132     * from <code>sequence</code> and starting at node <code>node</code>.
133     *
134     * @param sequence An array of labels to follow in the FSA.
135     * @param start    Starting index in <code>sequence</code>.
136     * @param length   How many symbols to consider from <code>sequence</code>?
137     * @param node     Start node identifier in the FSA.
138     */
139    public MatchResult match(byte[] sequence, int start, int length, int node) {
140        return match(new MatchResult(), sequence, start, length, node);
141    }
142
143    /**
144     * @see #match(byte[], int, int, int)
145     */
146    public MatchResult match(byte[] sequence, int node) {
147        return match(sequence, 0, sequence.length, node);
148    }
149
150    /**
151     * @see #match(byte[], int, int, int)
152     */
153    public MatchResult match(byte[] sequence) {
154        return match(sequence, fsa.getRootNode());
155    }
156}
