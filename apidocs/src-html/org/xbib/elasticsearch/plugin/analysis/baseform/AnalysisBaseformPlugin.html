001package org.xbib.elasticsearch.plugin.analysis.baseform;
002
003import org.elasticsearch.index.analysis.AnalysisModule;
004import org.elasticsearch.plugins.AbstractPlugin;
005import org.xbib.elasticsearch.index.analysis.baseform.BaseformTokenFilterFactory;
006
007public class AnalysisBaseformPlugin extends AbstractPlugin {
008
009    @Override
010    public String name() {
011        return "analysis-baseform-" +
012                Build.getInstance().getVersion() + "-" +
013                Build.getInstance().getShortHash();
014    }
015
016    @Override
017    public String description() {
018        return "A baseform token filter for german and other languages";
019    }
020
021    public void onModule(AnalysisModule module) {
022        module.addTokenFilter("baseform", BaseformTokenFilterFactory.class);
023    }
024}
