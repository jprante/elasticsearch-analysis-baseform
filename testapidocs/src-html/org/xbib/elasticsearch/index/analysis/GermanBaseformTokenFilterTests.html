001package org.xbib.elasticsearch.index.analysis;
002
003import java.io.IOException;
004
005import org.apache.lucene.analysis.TokenStream;
006import org.apache.lucene.analysis.tokenattributes.CharTermAttribute;
007
008import org.elasticsearch.common.inject.Injector;
009import org.elasticsearch.common.inject.ModulesBuilder;
010import org.elasticsearch.common.settings.ImmutableSettings;
011import org.elasticsearch.common.settings.Settings;
012import org.elasticsearch.common.settings.SettingsModule;
013import org.elasticsearch.env.Environment;
014import org.elasticsearch.env.EnvironmentModule;
015import org.elasticsearch.index.Index;
016import org.elasticsearch.index.IndexNameModule;
017import org.elasticsearch.index.analysis.AnalysisModule;
018import org.elasticsearch.index.analysis.AnalysisService;
019import org.elasticsearch.index.analysis.NamedAnalyzer;
020import org.elasticsearch.index.settings.IndexSettingsModule;
021import org.elasticsearch.indices.analysis.IndicesAnalysisModule;
022import org.elasticsearch.indices.analysis.IndicesAnalysisService;
023
024import org.testng.Assert;
025import org.testng.annotations.BeforeClass;
026import org.testng.annotations.Test;
027import org.xbib.elasticsearch.plugin.analysis.baseform.AnalysisBaseformPlugin;
028
029public class GermanBaseformTokenFilterTests extends Assert {
030
031    NamedAnalyzer analyzer;
032
033    @BeforeClass
034    public void create() {
035        AnalysisService analysisService = createAnalysisService();
036        analyzer = analysisService.analyzer("baseform");
037    }
038
039    @Test
040    public void test1() throws IOException {
041
042        String source = "Die Jahresfeier der Rechtsanwaltskanzleien auf dem Donaudampfschiff hat viel Ökosteuer gekostet";
043
044        String[] expected = {
045            "Die",
046            "Jahresfeier",
047            "der",
048            "Rechtsanwaltskanzleien",
049            "Rechtsanwaltskanzlei",
050            "auf",
051            "dem",
052            "Donaudampfschiff",
053            "hat",
054            "haben",
055            "viel",
056            "Ökosteuer",
057            "gekostet",
058            "kosten"
059        };
060        assertSimpleTSOutput(analyzer.tokenStream("content", source), expected);
061    }
062
063    @Test
064    public void test2() throws IOException {
065        String source = "Das sind Autos, die Nudeln transportieren.";
066        String[] expected = {
067                "Das",
068                "sind",
069                "Autos",
070                "Auto",
071                "die",
072                "der",
073                "Nudeln",
074                "Nudel",
075                "transportieren"
076        };
077        assertSimpleTSOutput(analyzer.tokenStream("content", source), expected);
078    }
079
080    @Test
081    public void test3() throws IOException {
082        String source = "dieser test dauert lange";
083        String[] expected = {
084                "dieser",
085                "dies",
086                "test",
087                "dauert",
088                "dauern",
089                "lange",
090                "langen"
091        };
092        assertSimpleTSOutput(analyzer.tokenStream("content", source), expected);
093    }
094
095    private AnalysisService createAnalysisService() {
096        Settings settings = ImmutableSettings.settingsBuilder()
097                .loadFromClasspath("org/xbib/elasticsearch/index/analysis/baseform_de.json").build();
098
099        Index index = new Index("test");
100
101        Injector parentInjector = new ModulesBuilder().add(new SettingsModule(settings),
102                new EnvironmentModule(new Environment(settings)),
103                new IndicesAnalysisModule())
104                .createInjector();
105
106        AnalysisModule analysisModule = new AnalysisModule(settings, parentInjector.getInstance(IndicesAnalysisService.class));
107        new AnalysisBaseformPlugin().onModule(analysisModule);
108
109        Injector injector = new ModulesBuilder().add(
110                new IndexSettingsModule(index, settings),
111                new IndexNameModule(index),
112                analysisModule)
113                .createChildInjector(parentInjector);
114
115        return injector.getInstance(AnalysisService.class);
116    }
117
118    private void assertSimpleTSOutput(TokenStream stream, String[] expected) throws IOException {
119        stream.reset();
120        CharTermAttribute termAttr = stream.getAttribute(CharTermAttribute.class);
121        assertNotNull(termAttr);
122        int i = 0;
123        while (stream.incrementToken()) {
124            assertTrue(i < expected.length);
125            assertEquals(expected[i++], termAttr.toString(), "expected different term at index " + i);
126        }
127        assertEquals(i, expected.length, "not all tokens produced");
128        stream.close();
129    }
130}
