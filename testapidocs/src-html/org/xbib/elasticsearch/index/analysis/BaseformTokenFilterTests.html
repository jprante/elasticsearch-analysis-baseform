037    @Test
038    public void testOne() throws IOException {
039        AnalysisService analysisService = createAnalysisService();
040        TokenFilterFactory tokenFilter = analysisService.tokenFilter("baseform");
041        assertThat(tokenFilter, instanceOf(BaseformTokenFilterFactory.class));
042
043        String source = "Die Jahresfeier der Rechtsanwaltskanzleien auf dem Donaudampfschiff hat viel Ökosteuer gekostet";
044
045        String[] expected = {
046            "Die",
047            "Die",
048            "Jahresfeier",
049            "Jahresfeier",
050            "der",
051            "der",
052            "Rechtsanwaltskanzleien",
053            "Rechtsanwaltskanzlei",
054            "auf",
055            "auf",
056            "dem",
057            "der",
058            "Donaudampfschiff",
059            "Donaudampfschiff",
060            "hat",
061            "haben",
062            "viel",
063            "viel",
064            "Ökosteuer",
065            "Ökosteuer",
066            "gekostet",
067            "kosten"
068        };
069
070        Tokenizer tokenizer = new StandardTokenizer(Version.LUCENE_44, new StringReader(source));
071        assertSimpleTSOutput(tokenFilter.create(tokenizer), expected);
072
073    }
074
075    @Test
076    public void testTwo() throws IOException {
077        AnalysisService analysisService = createAnalysisService();
078        TokenFilterFactory tokenFilter = analysisService.tokenFilter("baseform");
079        assertThat(tokenFilter, instanceOf(BaseformTokenFilterFactory.class));
080
081        String source = "Das sind Autos, die Nudeln transportieren.";
082
083        String[] expected = {
084                "Das",
085                "Das",
086                "sind",
087                "sind",
088                "Autos",
089                "Auto",
090                "die",
091                "der",
092                "Nudeln",
093                "Nudel",
094                "transportieren",
095                "transportieren"
096        };
097
098        Tokenizer tokenizer = new StandardTokenizer(Version.LUCENE_44, new StringReader(source));
099        assertSimpleTSOutput(tokenFilter.create(tokenizer), expected);
100    }
101
102    public AnalysisService createAnalysisService() {
103        Settings settings = ImmutableSettings.settingsBuilder()
104                .loadFromClasspath("org/xbib/elasticsearch/index/analysis/baseform_de.json").build();
105
106        Index index = new Index("test");
107
108        Injector parentInjector = new ModulesBuilder().add(new SettingsModule(settings),
109                new EnvironmentModule(new Environment(settings)),
110                new IndicesAnalysisModule())
111                .createInjector();
112
113        AnalysisModule analysisModule = new AnalysisModule(settings, parentInjector.getInstance(IndicesAnalysisService.class));
114        new AnalysisBaseformPlugin().onModule(analysisModule);
115
116        Injector injector = new ModulesBuilder().add(
117                new IndexSettingsModule(index, settings),
118                new IndexNameModule(index),
119                analysisModule)
120                .createChildInjector(parentInjector);
121
122        return injector.getInstance(AnalysisService.class);
123    }
124
125    public static void assertSimpleTSOutput(TokenStream stream,
126            String[] expected) throws IOException {
127        stream.reset();
128        CharTermAttribute termAttr = stream.getAttribute(CharTermAttribute.class);
129        assertNotNull(termAttr);
130        int i = 0;
131        while (stream.incrementToken()) {
132            assertTrue(i < expected.length);
133            assertEquals(expected[i++], termAttr.toString(), "expected different term at index " + i);
134        }
135        assertEquals(i, expected.length, "not all tokens produced");
136    }
